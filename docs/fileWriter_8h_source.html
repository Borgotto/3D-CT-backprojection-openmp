58bool writeVolumeNRRD(FILE* file, volume* volume) {
59 // Write the NRRD header
60 fprintf(file, "NRRD0005\n");

62 fprintf(file, "# http://teem.sourceforge.net/nrrd/format.html\n");

64 fprintf(file, "dimension: 3\n");
65 fprintf(file, "sizes: %d %d %d\n",
66 volume->nVoxelsX, volume->nVoxelsY, volume->nVoxelsZ);
67 fprintf(file, "spacings: %g %g %g\n",
68 volume->voxelSizeX, volume->voxelSizeY, volume->voxelSizeZ);
69 fprintf(file, "axis mins: %g %g %g\n",
70 -volume->nVoxelsX * volume->voxelSizeX / 2,
71 -volume->nVoxelsY * volume->voxelSizeY / 2,
72 -volume->nVoxelsZ * volume->voxelSizeZ / 2);
73
74 // Set the endianness of the data
75 #if !defined(__ORDER_BIG_ENDIAN__) && !defined(__ORDER_LITTLE_ENDIAN__)
76 #error "Endianness not defined"
77 #elif __FLOAT_WORD_ORDER__ == __ORDER_LITTLE_ENDIAN__
78 fprintf(file, "endian: little\n");
79 #else
80 fprintf(file, "endian: big\n");
81 #endif
82
83 // Set the encoding of the data and write the coefficients
84 #ifdef _OUTPUT_FORMAT_ASCII
85 fprintf(file, "encoding: ascii\n\n");
86 for (int i = 0; i < volume->nVoxelsX * volume->nVoxelsY * volume->nVoxelsZ; i++) {
87 fprintf(file, "%g ", volume->coefficients[i]);
88 }
89 #else
90 fprintf(file, "encoding: raw\n\n");
91 size_t numVoxels = volume->nVoxelsX * volume->nVoxelsY * volume->nVoxelsZ;
92 if (fwrite(volume->coefficients, sizeof(double), numVoxels, file) != numVoxels) {
93 return false; // If fwrite doesn't write all the data, return false
94 }
95 #endif
96
97
98 return true;
99}
