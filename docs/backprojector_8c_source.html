45void initTables() {
46 // Calculate sin and cos values for the angle of this projection
47 for (int i = 0; i < N_THETA; i++) {
48 const long double angle = AP / 2 + i * STEP_ANGLE;
49 const long double rad = angle * M_PI / 180;
50 sinTable[i] = sinl(rad);
51 cosTable[i] = cosl(rad);
52 }
53
54 // Calculate the first and last plane positions for each axis
55 for (axis axis = X; axis <= Z; axis++) {
56 firstPlane[axis] = -(VOXEL_SIZE[axis] * N_VOXELS[axis]) / 2;
57 lastPlane[axis] = -firstPlane[axis];
58 }
59}

69point3D getPixelPosition(const projection* projection, const int row, const int col) {
70 // This is the distance from the center of the detector to the top-left pixel's center
71 // it's used to calculate the center position of subsequent pixels
72 const double dFirstPixel = projection->nSidePixels * PIXEL_SIZE / 2 - PIXEL_SIZE / 2;
73 const double sinAngle = sinTable[projection->index];
74 const double cosAngle = cosTable[projection->index];
75
76 return (point3D) {
77 .x = DOD * sinAngle + cosAngle * (-dFirstPixel + col * PIXEL_SIZE),
78 .y = -DOD * cosAngle + sinAngle * (-dFirstPixel + col * PIXEL_SIZE),
79 .z = -dFirstPixel + row * PIXEL_SIZE
80 };

101void getSidesIntersections(const ray ray, const axis parallelTo, double intersections[3][2]) {
102 const point3D source = ray.source;
103 const point3D pixel = ray.pixel;
104 for (axis axis = X; axis <= Z; axis++) {
105 if (axis == parallelTo) {
106 continue; // Skip the axis that the ray is parallel to
107 }
108
109 // Calculate the entry and exit points of the ray with the planes of this axis
110 // Siddon's algorithm, equation (4)
111 intersections[axis][0] = (firstPlane[axis] - source.coordinates[axis]) / (pixel.coordinates[axis] - source.coordinates[axis]);
112 intersections[axis][1] = (lastPlane[axis] - source.coordinates[axis]) / (pixel.coordinates[axis] - source.coordinates[axis]);
113 }
114}

116double getAMin(const axis parallelTo, double intersections[3][2]) {
117 double aMin = 0.0;

142void getPlanesRanges(const ray ray, range planesRanges[3], const double aMin, const double aMax) {
143 const point3D source = ray.source;
144 const point3D pixel = ray.pixel;
145 for (axis axis = X; axis <= Z; axis++) {
146 // Siddon's algorithm, equation (6)
147 int minIndex, maxIndex;
148 if (pixel.coordinates[axis] - source.coordinates[axis] >= 0) {
149 minIndex = N_PLANES[axis] - ceil((lastPlane[axis] - aMin * (pixel.coordinates[axis] - source.coordinates[axis]) - source.coordinates[axis]) / VOXEL_SIZE[axis]);
150 maxIndex = floor((source.coordinates[axis] + aMax * (pixel.coordinates[axis] - source.coordinates[axis]) - firstPlane[axis]) / VOXEL_SIZE[axis]);
151 } else {
152 minIndex = N_PLANES[axis] - ceil((lastPlane[axis] - aMax * (pixel.coordinates[axis] - source.coordinates[axis]) - source.coordinates[axis]) / VOXEL_SIZE[axis]);
153 maxIndex = floor((source.coordinates[axis] + aMin * (pixel.coordinates[axis] - source.coordinates[axis]) - firstPlane[axis]) / VOXEL_SIZE[axis]);
154 }
155 planesRanges[axis] = (range){.min=minIndex, .max=maxIndex};
156 }

159void getAllIntersections(const ray ray, const range planesRanges[3], double* a[3]) {
160 const point3D source = ray.source;
161 const point3D pixel = ray.pixel;
162 for (axis axis = X; axis <= Z; axis++) {
163 int minIndex = planesRanges[axis].min;
164 int maxIndex = planesRanges[axis].max;
165
166 if (minIndex >= maxIndex) {
167 continue;
168 }
169
170 #ifdef _DEBUG
171 assert(pixel.coordinates[axis] - source.coordinates[axis] != 0);
172 #endif
173
174 // Siddon's algorithm, equation (7)
175 if (pixel.coordinates[axis] - source.coordinates[axis] > 0) {
176 a[axis][0] = (getPlanePosition(axis, minIndex) - source.coordinates[axis]) / (pixel.coordinates[axis] - source.coordinates[axis]);
177 for (int i = 1; i < maxIndex - minIndex; i++) {
178 a[axis][i] = a[axis][i - 1] + VOXEL_SIZE[axis] / (pixel.coordinates[axis] - source.coordinates[axis]);
179 }
180 } else if (pixel.coordinates[axis] - source.coordinates[axis] < 0) {
181 a[axis][0] = (getPlanePosition(axis, maxIndex) - source.coordinates[axis]) / (pixel.coordinates[axis] - source.coordinates[axis]);
182 for (int i = 1; i < maxIndex - minIndex; i++) {
183 // TODO: check if '+' or '-' is correct here (it's '+' in Siddon's algorithm, but '-' seems to be correct)
184 a[axis][i] = a[axis][i - 1] - VOXEL_SIZE[axis] / (pixel.coordinates[axis] - source.coordinates[axis]);
185 }
186 }
187 }

239void computeAbsorption(const ray ray, const double a[], const int lenA, const double pixelAbsorptionValue, double* coefficients) {
240 const point3D source = ray.source;
241 const point3D pixel = ray.pixel;
242
243 // distance between the source and the pixel
244 // Siddon's algorithm, equation (11)
245 const double dx = pixel.x - source.x;
246 const double dy = pixel.y - source.y;
247 const double dz = pixel.z - source.z;
248 const double d12 = sqrt(dx * dx + dy * dy + dz * dz);
249
250 // TODO: this needs to be optimized, see if it's possible to use SIMD instructions
251 for(int i = 1; i < lenA; i ++){
252 // Siddon's algorithm, equation (10)
253 const double segmentLength = d12 * (a[i] - a[i-1]);
254 // Siddon's algorithm, equation (13)
255 const double aMid = (a[i] + a[i-1]) / 2;
256
257 // Calculate the voxel indices that the ray intersects
258 // Siddon's algorithm, equation (12)
259 const int voxelX = (source.x + aMid * (pixel.x - source.x) - firstPlane[X]) / VOXEL_SIZE_X;
260 const int voxelY = (source.y + aMid * (pixel.y - source.y) - firstPlane[Y]) / VOXEL_SIZE_Y;
261 const int voxelZ = (source.z + aMid * (pixel.z - source.z) - firstPlane[Z]) / VOXEL_SIZE_Z;
262
263 // Update the value of the voxel given the value of the pixel and the
264 // length of the segment that the ray intersects with the voxel
265 const double voxelAbsorptionValue = pixelAbsorptionValue * segmentLength;
266 const int voxelIndex = voxelY * N_VOXELS_X * N_VOXELS_Z + voxelZ * N_VOXELS_Z + voxelX;
267
268 #ifdef _DEBUG
269 assert(voxelX >= 0 && voxelY >= 0 && voxelZ >= 0);
270 assert(voxelX <= N_VOXELS_X && voxelY <= N_VOXELS_Y && voxelZ <= N_VOXELS_Z);
271 assert(voxelAbsorptionValue >= 0);
272 assert(voxelIndex >= 0 && voxelIndex <= N_VOXELS_X * N_VOXELS_Y * N_VOXELS_Z);
273 #endif
274
275 // TODO: find a way to avoid using atomic operations, this is a bottleneck
276 // Siddon's algorithm, equation (14)
277 #pragma omp atomic update
278 coefficients[voxelIndex] += voxelAbsorptionValue;
279 }

282void computeBackProjection(const projection* projection, volume* volume) {
283 // Check if the arguments are valid
284 if (volume == NULL || projection == NULL) {
285 fprintf(stderr, "Invalid arguments\n");
286 exit(1);
287 }
288
289 // Get the source point of this projection
290 const point3D source = getSourcePosition(projection->index);
291
292 // Iterate through every pixel of the projection image and calculate the
293 // coefficients of the voxels that contribute to the pixel.
294 //#pragma omp parallel for collapse(2) schedule(dynamic) default(none) shared(volume, projection, source)
295 // TODO: to reduce the atomic operations, it may be possible to offset the pixel processing by a certain amount * thread_id
296 // TODO: so that each thread shoots rays at different pixels, this could reduce the intersecting voxels between threads
297 for (int row = 0; row < projection->nSidePixels; row++) {
298 for (int col = 0; col < projection->nSidePixels; col++) {
299 const point3D pixel = getPixelPosition(projection, row, col);
300 const ray ray = {.source=source, .pixel=pixel};
301 const axis parallelTo = getParallelAxis(ray);
302
303 // This array will contain the intersection points of the ray with
304 // the planes. For each (3) axis the first element is the entry point
305 // of the ray into the first plane of that axis and the second element
306 // is the exit point of the ray from the last plane of that axis.
307 double intersections[3][2];
308 getSidesIntersections(ray, parallelTo, intersections);
309
310 // Find aMin and aMax with intersections with the side planes
311 double aMin = getAMin(parallelTo, intersections);
312 double aMax = getAMax(parallelTo, intersections);
313
314 if (aMin >= aMax) {
315 continue; // The ray doesn't intersect the volume
316 }
317
318 /*
319 * These are the (min, max) indices of the planes that the ray intersects
320 * in the x, y, and z axes. These indices are used to calculate the
321 * coefficients of the voxels that the ray intersects.
322 */
323 range planesRanges[3];
324 getPlanesRanges(ray, planesRanges, aMin, aMax);
325
326 #ifdef _DEBUG
327 assert(planesRanges[X].min >= 0 && planesRanges[Y].min >= 0 && planesRanges[Z].min >= 0);
328 assert(planesRanges[X].max <= N_PLANES_X && planesRanges[Y].max <= N_PLANES_Y && planesRanges[Z].max <= N_PLANES_Z);
329 #endif
330
331 // Calculate all of the intersections of the ray with the planes of
332 // all axes and merge them into a single array
333 const int aXSize = fmax(0, planesRanges[X].max - planesRanges[X].min);
334 const int aYSize = fmax(0, planesRanges[Y].max - planesRanges[Y].min);
335 const int aZSize = fmax(0, planesRanges[Z].max - planesRanges[Z].min);
336 double aX[aXSize], aY[aYSize], aZ[aZSize];
337 getAllIntersections(ray, planesRanges, (double*[]){aX, aY, aZ});
338
339 // Calculate the size of the merged array
340 // Siddon's algorithm, equation (9)
341 const int mergedSize = aXSize + aYSize + aZSize;
342 double aMerged[mergedSize];
343
344 // Merge all of the intersections into a single sorted array
345 // Siddon's algorithm, equation (8)
346 mergeIntersections(aX, aY, aZ, aXSize, aYSize, aZSize, aMerged);
347 #ifdef _DEBUG
348 assert(isArraySorted(aMerged, mergedSize));
349 #endif
350
351 // Calculate the coefficients of the voxels that the ray intersects
352 // TODO: finish implementing this properly
353 const double pixelAbsorptionValue = projection->pixels[row * projection->nSidePixels + col];
354 computeAbsorption(ray, aMerged, mergedSize, pixelAbsorptionValue, volume->coefficients);
355 }
356 }

361 // Check if input file is provided either as an argument or through stdin
362 if (argc < 2 && isatty(fileno(stdin))) { // TODO: find alternative to isatty
363 fprintf(stderr, "No input file provided\n");

369 if (inputFile == NULL) {
370 fprintf(stderr, "Error opening input file\n");
371 exit(1);
372 }
373
374 // Do the same with the output file
375 FILE* outputFile = (argc >= 3) ? fopen(argv[2], "wb") : stdout;
376 if (outputFile == NULL) {
377 fprintf(stderr, "Error opening output file\n");
378 exit(1);

382 /*
383 * This struct is used to store the calculated coefficients of the voxels
384 * using the backprojection algorithm starting from the projection images
385 */
386 volume volume = {
387 .nVoxelsX = N_VOXELS_X,
388 .nVoxelsY = N_VOXELS_Y,
389 .nVoxelsZ = N_VOXELS_Z,
390 .voxelSizeX = VOXEL_SIZE_X,
391 .voxelSizeY = VOXEL_SIZE_Y,
392 .voxelSizeZ = VOXEL_SIZE_Z,
393 // TODO: find a way to split this array into multiple parts because it's too big (1000^3)*8 bytes = 8GB
394 // TODO: a possible solution is to split the volume into chunks and process them serially, this reduces the memory usage but increases the runtime
395 // TODO: or sacrifice accuracy for memory by using float instead of double
396 .coefficients = (double*)calloc(N_VOXELS_X * N_VOXELS_Y * N_VOXELS_Z, sizeof(double))
397 };
398 // Check if the memory was allocated successfully
399 if (volume.coefficients == NULL) {
400 fprintf(stderr, "Error allocating memory for the volume\n");
401 exit(1);
402 }
403
404 initTables();
405
406 double initialTime = omp_get_wtime();
407 int width, height;
408 double maxVal;
409
410 // Read the projection images from the file and compute the backprojection
411 #pragma omp parallel for default(none) shared(inputFile, volume, width, height, maxVal, stderr)
412 for (int i = 0; i < N_THETA; i++) {
413 projection projection;
414 bool read;
415
416 // File reading has to be done sequentially
417 #pragma omp critical
418 read = readPGM(inputFile, &projection, &width, &height, &maxVal);
419
420 // if read is false, it means that the end of the file was reached
421 if (read) {
422 computeBackProjection(&projection, &volume);
423 }
424
425 free(projection.pixels);
426 }
427
428 double finalTime = omp_get_wtime();
429 fprintf(stderr, "%d: %lf\n", width, (finalTime - initialTime));
430
431 writeVolume(outputFile, &volume);
432
433 fclose(inputFile);
434 fclose(outputFile);
435 free(volume.coefficients);
436 return 0;
437}
